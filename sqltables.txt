
DEPARTEMENT(NOM,  BUDGET)

PROJET(NOM, DEPARTEMENT,DATE_DEBUT,CHEF,BUDGET,
            COUT,DATE_FIN)
o DEPARTEMENT → NOM de DEPARTEMENT, CHEF → NO de EMPLOYE

EMPLOYE(NO,  NOM,NOM_DEPARTEMENT,NOM_FONCTION)
o NOM_DEPARTEMENT → NOM de DEPARTEMENT,
o NOM_FONCTION → NOM de FONCTION

TACHE(EMPLOYE,PROJET,NOMBRE_HEURES)
o EMPLOYE → NO de EMPLOYE,
o PROJET → NOM de PROJET


EVALUATION(PROJET,  EXPERT,COMMENTAIRES,AVIS)
o PROJET → NOM de PROJET, EXPERT → NO de EMPLOYE

FONCTION(NOM,  TAUX_HORAIRE)
MOTS_CLES(RAPPORT,MOT_CLE)
o RAPPORT → TITRE de RAPPORT


RAPPORT(EMPLOYE, PROJET,  TITRE)
o EMPLOYE/PROJET → EMPLOYE/PROJET de TACHE
SELECT EMPLOYE.NO, EMPLOYE.NOM, (EMPLOYE.NOM_FONCTION) AS FONCTION,
       GROUP_CONCAT(DISTINCT PROJET.NOM SEPARATOR " et ") AS CHEF_DE ,
       GROUP_CONCAT(DISTINCT TACHE.PROJET SEPARATOR " et ") AS EMPLOYE_POUR,
       GROUP_CONCAT(DISTINCT EVALUATION.PROJET SEPARATOR " et ") AS EXPERT_POUR
FROM EMPLOYE LEFT JOIN TACHE ON EMPLOYE.NO = TACHE.EMPLOYE
             LEFT JOIN PROJET ON EMPLOYE.NO = PROJET.CHEF
             LEFT JOIN EVALUATION ON EMPLOYE.NO = EVALUATION.EXPERT
WHERE (SELECT COUNT(DISTINCT PROJET.NOM) FROM PROJET) = (SELECT COUNT(DISTINCT PROJET.NOM)) + (SELECT COUNT(DISTINCT TACHE.PROJET)) + (SELECT COUNT(DISTINCT EVALUATION.PROJET))
GROUP BY EMPLOYE.NO
